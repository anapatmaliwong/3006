  // 1. Listen for auth tokens sent via postMessage (from the login popup window)
  window.addEventListener('message', async function(event) {
    var allowedOrigins = ['https://headface.app', 'https://beta.headface.app'];
    var isAllowed = allowedOrigins.indexOf(event.origin) !== -1 || event.origin.includes('localhost') || event.origin.includes('127.0.0.1');
    if (!isAllowed) return;

    if (event.data && (event.data.type === 'firebase-token' || event.data.type === 'google-code' || event.data.type === 'auth-code')) {
      var authCode = event.data.code || event.data.token;
      if (!authCode) return;

      try {
        console.log("Fetching student profile from Workers API using code...");
        var res = await fetch("https://headface.ac-headface.workers.dev/login", {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ code: authCode, OAuth: true })
        });
        var json = await res.json();
        
        if (json && json.data && json.data.length > 0) {
          var data = json.data[0];
          var firebaseToken = data.token || json.token || data.firebaseToken || "";

          // Save student profile data to localStorage
          localStorage.setItem("first_name", data.first_name);
          localStorage.setItem("last_name", data.last_name);
          localStorage.setItem("number", data.number);
          localStorage.setItem("student_id", data.student_id);
          localStorage.setItem("token", firebaseToken);
          console.log("Profile data saved successfully.");

          // Log in to Firebase Auth using the returned firebaseToken
          if (firebaseToken) {
            var isIdToken = false;
            var payload = null;
            try {
              var base64Url = firebaseToken.split('.')[1];
              var base64 = base64Url.replace(/-/g, '+').replace(/_/g, '/');
              var jsonPayload = decodeURIComponent(atob(base64).split('').map(function(c) {
                return '%' + ('00' + c.charCodeAt(0).toString(16)).slice(-2);
              }).join(''));
              payload = JSON.parse(jsonPayload);
              if (payload && payload.iss && payload.iss.indexOf('securetoken.google.com') !== -1) {
                isIdToken = true;
              }
            } catch (e) {}

            if (isIdToken && payload) {
              var apiKey = "AIzaSyAK7hHbowArjD7tQlinwgItiplQ21C8vkk";
              var key = "firebase:authUser:" + apiKey + ":[DEFAULT]";
              var authUserData = {
                uid: payload.user_id || payload.sub,
                email: payload.email,
                emailVerified: payload.email_verified || true,
                displayName: payload.name || "",
                photoURL: payload.picture || "",
                isAnonymous: false,
                providerData: [{
                  providerId: "google.com",
                  uid: payload.user_id || payload.sub,
                  displayName: payload.name || "",
                  email: payload.email,
                  photoURL: payload.picture || ""
                }],
                stsTokenManager: {
                  apiKey: apiKey,
                  refreshToken: "",
                  accessToken: firebaseToken,
                  expirationTime: (payload.exp * 1000)
                },
                apiKey: apiKey,
                appName: "[DEFAULT]"
              };
              localStorage.setItem(key, JSON.stringify(authUserData));
              console.log("Firebase Auth session stored manually.");
              window.location.reload();
            } else {
              console.log("Signing in with Custom Token via SDK...");
              firebase.auth().signInWithCustomToken(firebaseToken).then(function() {
                console.log("Firebase Custom Token sign-in success.");
                window.location.reload();
              }).catch(function(err) {
                console.error("signInWithCustomToken failed:", err);
                window.location.reload();
              });
            }
          } else {
            console.warn("No firebase token returned from API to sign in with.");
            window.location.reload();
          }
        } else {
          console.warn("No student profile data found in worker API response");
          window.location.reload();
        }
      } catch (err) {
        console.error("Failed to authenticate and store token:", err);
      }
    }
  });
